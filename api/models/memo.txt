-- userテーブルの作成
CREATE TABLE `user` (
    `id` CHAR(36) NOT NULL PRIMARY KEY,
    `name` VARCHAR(10),
    `email` VARCHAR(30),
    `password` VARCHAR(64),
    `comment` VARCHAR(100)
);

-- postテーブルの作成
CREATE TABLE `post` (
    `id` CHAR(36) NOT NULL PRIMARY KEY,
    `user_id` CHAR(36) NOT NULL,
    `title` VARCHAR(100),
    `content` VARCHAR(1000),
    FOREIGN KEY (`user_id`) REFERENCES `user` (`id`) ON DELETE CASCADE
);
